package problem1;

import java.util.Arrays;
import java.util.List;

public class PhoneNumbers {

  public PhoneNumbers() {
  }

  public String convertPhone(List<String> list) {
    List<String> individuals = null;
    for (String s : list) {
      String[] each = s.split("");
      individuals.add(Arrays.toString(String.valueOf(each[3]).split(":")));
      System.out.println(individuals);
    }
    return "no";
  }
}
